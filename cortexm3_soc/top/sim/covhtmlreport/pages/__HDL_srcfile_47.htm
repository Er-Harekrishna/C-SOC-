     8  /**
     9   * Abstract:
     10   * This class defines a sequence in which a INCR busrt with maximum busy cycles 
     11   * and maximum wait cycles are inserted to AHB WRITE followed by a random AHB READ
     12   * sequence is generated using `uvm_do_with macros.
     13   *
     14   * Execution phase: main_phase
     15   * Sequencer: ahb_mas_seqr
     16   *
     17   */
     18  	 
     19  `ifndef AHB_WAITED_MAX_BUSY_SEQUENCE
     20  `define AHB_WAITED_MAX_BUSY_SEQUENCE
     21  
     22  class ahb_master_waited_max_busy_sequence extends ahb_base_master_sequence;
     23    
     24    /** factroy registration */
     25    `uvm_object_utils(ahb_master_waited_max_busy_sequence)
     26   
     27    /** class constructor */
     28    extern function new(string name="ahb_master_waited_max_busy_sequence");
     29    
     30    /** body method for generating stimulus */
     31    extern task body();
     32  
     33  endclass : ahb_master_waited_max_busy_sequence
     34  
     35  //*****************************************************************************
     36  //methods
     37  //*****************************************************************************
     38  
     39  /** class constructor */
     40  function ahb_master_waited_max_busy_sequence::new(string name="ahb_master_waited_max_busy_sequence");
     41    super.new(name);
     42  endfunction : new
     43    
     44  /** body method for generating stimulus */
     45  task ahb_master_waited_max_busy_sequence::body();
     46    
     47    `uvm_info("body", "Entered ...", UVM_MEDIUM)
     48  
     49    if(!(uvm_config_db #(int unsigned)::get(null, get_full_name(), "sequence_length", sequence_length)))
     50      `uvm_info("BODY","GET FAILED",UVM_MEDIUM)
     51  
     52    `uvm_info("body", $sformatf("sequence_length is %0d", sequence_length), UVM_MEDIUM);
     53    
     54    for(int k=0;k<8;k++) begin
     55  
     56      for(int i=0;i<=$clog2(`DATA_WIDTH/8);i++) begin
     57  
     58        `uvm_do_with(req,
     59         { HADDR       == 2**(i);
     60           HWRITE      == 1;
     61   	 hburst_type == hburst_enum'(k);
     62  	 hsize_type  == hsize_enum'(i);
     63  	 incr_size   == 3;
     64  	 if(beat_cnt!=1) busy_trans_cycles[0]==16;
     65         })
     66      
     67        `uvm_do_with(req,
     68         { HADDR       == 2**(i);
     69           HWRITE      == 0;
     70   	 hburst_type == hburst_enum'(k);
     71  	 hsize_type  == hsize_enum'(i);
     72  	 incr_size   == 3;
     73  	 if(beat_cnt!=1) busy_trans_cycles[0]==16;
     74         })
     75  
     76      end
     77    end
     78  
     79    `uvm_info("body", "Exiting ...", UVM_MEDIUM)
     80  
     81  endtask : body
     82  
     83  `endif //AHB_WAITED_MAX_BUSY_SEQUENCE
