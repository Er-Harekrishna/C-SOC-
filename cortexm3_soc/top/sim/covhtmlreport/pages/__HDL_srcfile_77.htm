     77  `timescale 1ns/1ps //timescale 
     78  `include "fifo.v"
     79  `include "module_i2c.v"
     80  `include "apb.v"
     81  module i2c(
     82  	//APB PORTS
     83  	input PCLK,
     84  	input PRESETn,
     85  	input [31:0] PADDR,
     86  	input [31:0] PWDATA,
     87  	input PWRITE,
     88  	input PSELx,
     89  	input PENABLE,
     90  	output PREADY,
     91  	output PSLVERR,
     92  	output INT_RX,
     93  	output INT_TX,
     94  	output [31:0] PRDATA,
     95  	//I2C OUTPUT
     96  	output SDA_ENABLE,
     97  	output SCL_ENABLE,
     98  	inout SDA,
     99  	inout SCL
     100  
     101  	  );
     102  
     103  	wire RESET_N;
     104  
     105  	//THIS IS USED TO RESET FIFO
     106  	assign RESET_N = (PRESETn == 0)?1'b1:1'b0;	
     107  
     108  	//WIRES USED TO CONECT BLOCK WITH EACH OTHER
     109  	wire TX_RD_EN;
     110  	wire TX_F_EMPTY;
     111  	wire TX_F_FULL;
     112  	wire [31:0] TX_DATA_IN;
     113  	wire [31:0] TX_DATA_OUT;
     114  	wire TX_WRITE_ENA;
     115  
     116  	wire RX_RD_EN;
     117  	wire RX_F_EMPTY;
     118  	wire RX_F_FULL;
     119  	wire [31:0] RX_DATA_IN;
     120  	wire [31:0] RX_DATA_OUT;
     121  	wire RX_WRITE_ENA;
     122  
     123  
     124  	wire [13:0] REGISTER_CONFIG;
     125  	wire [13:0] TIMEOUT_CONFIG;
     126  
     127  
     128  	wire error;
     129  
     130  
     131  	wire tx_empty;
     132  	wire rx_empty;
     133  
     134  	//wire w_pwrite;
     135  	wire w_full;
     136  	//wire w_full_tx;
     137  
     138  	//assign w_pwrite = (PWRITE == 1'b0)?1'b1:1'b0;
     139  		
     140  	
     141  
     142  	//CONECTIONS WITH FIFO TX
     143  	fifo DUT_FIFO_TX (
     144  				.clock(PCLK),
     145  				.reset(RESET_N),
     146  				.wr_en(TX_WRITE_ENA), 
     147  				.rd_en(TX_RD_EN),
     148  				.data_in(TX_DATA_IN),
     149  				.f_full(w_full), 
     150  				.f_empty(TX_F_EMPTY),
     151  				.data_out(TX_DATA_OUT)
     152  		         );
     153  
     154  
     155  	//and(w_full_tx,w_pwrite,w_full);
     156  
     157  	assign TX_F_FULL = w_full;
     158  
     159  	//CONECTIONS WITH FIFO RX
     160  	fifo DUT_FIFO_RX (
     161  				.clock(PCLK),
     162  				.reset(RESET_N),
     163  				.wr_en(RX_WRITE_ENA), 
     164  				.rd_en(RX_RD_EN),
     165  				.data_in(RX_DATA_IN),
     166  				.f_full(RX_F_FULL), 
     167  				.f_empty(RX_F_EMPTY),
     168  				.data_out(RX_DATA_OUT)
     169  		         );
     170  
     171  	//CONECTIONS WITH APB AND ALL BLOCKS WHERE IS TWO FIFOS AND I2C CORE
     172  	apb DUT_APB (
     173  
     174  			.PCLK(PCLK),
     175  			.PRESETn(PRESETn),
     176                          .PADDR(PADDR),
     177  			.PRDATA(PRDATA),
     178  			.PWDATA(PWDATA),
     179  			.PWRITE(PWRITE),
     180  			.PSELx(PSELx),
     181  			.PENABLE(PENABLE),
     182  			.PREADY(PREADY),
     183  			.PSLVERR(PSLVERR),
     184  			.READ_DATA_ON_RX(RX_DATA_OUT),
     185  			.INTERNAL_I2C_REGISTER_CONFIG(REGISTER_CONFIG),
     186  			.INTERNAL_I2C_REGISTER_TIMEOUT(TIMEOUT_CONFIG),
     187  			.INT_RX(INT_RX),
     188  			.WR_ENA(TX_WRITE_ENA),
     189  			.WRITE_DATA_ON_TX(TX_DATA_IN),
     190  			.RD_ENA(RX_RD_EN),
     191  			.INT_TX(INT_TX),
     192  			.TX_EMPTY(tx_empty),
     193  			.RX_EMPTY(rx_empty),
     194  			.ERROR(error)
     195  
     196  		     );
     197  
     198  	//I2C CORE BLOCK WITH ALL ANOTHER BLOCKS
     199  	module_i2c DUT_I2C_INTERNAL_RX_TX (
     200  				  	.PCLK(PCLK),
     201  					.PRESETn(PRESETn),
     202  					.fifo_rx_wr_en(RX_WRITE_ENA),
     203  					.fifo_rx_f_empty(RX_F_EMPTY),
     204  					.fifo_rx_data_in(RX_DATA_IN),
     205  					.fifo_rx_f_full(RX_F_FULL),
     206  					.fifo_tx_f_full(TX_F_FULL),
     207  					.fifo_tx_f_empty(TX_F_EMPTY),
     208  					.fifo_tx_rd_en(TX_RD_EN),
     209  					.fifo_tx_data_out(TX_DATA_OUT),
     210  					.DATA_CONFIG_REG(REGISTER_CONFIG),
     211  					.TIMEOUT_TX(TIMEOUT_CONFIG),
     212  					.RX_EMPTY(rx_empty),
     213  					.TX_EMPTY(tx_empty),					
     214  					.ERROR(error),
     215  					.ENABLE_SDA(SDA_ENABLE),
     216  					.ENABLE_SCL(SCL_ENABLE),
     217  					.SDA(SDA),
     218  					.SCL(SCL)
     219  				    );
     220  
     221  endmodule
