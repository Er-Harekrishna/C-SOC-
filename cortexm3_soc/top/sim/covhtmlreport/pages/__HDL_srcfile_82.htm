     29     INTISR, INTNMI, HREADYI, HRDATAI, HRESPI, IFLUSH, HREADYD, HRDATAD, HRESPD,
     30     EXRESPD, SE, HREADYS, HRDATAS, HRESPS, EXRESPS, EDBGRQ, DBGRESTART, RXEV,
     31     SLEEPHOLDREQn, WICENREQ, FIXMASTERTYPE, TSVALUEB, MPUDISABLE,
     32     DBGEN, NIDEN, CDBGPWRUPACK, DNOTITRANS,
     33     // Outputs
     34     TDO, nTDOEN, SWDOEN, SWDO, SWV, JTAGNSW, TRACECLK, TRACEDATA, HTRANSI,
     35     HSIZEI, HADDRI, HBURSTI, HPROTI, MEMATTRI, HTRANSD, HSIZED, HADDRD, HBURSTD,

     47    //----------------------------------------------------------------------------
     48    // Port declarations
     49    //----------------------------------------------------------------------------
     50  
     51    // PMU
     52    input          ISOLATEn;           // Isolate core power domain
     53    input          RETAINn;            // Retain core state during power-down
     54  
     55    // Debug
     56    input          nTRST;              // Test reset
     57    input          SWDITMS;            // Test Mode Select/SWDIN
     58    input          SWCLKTCK;           // Test clock / SWCLK
     59    input          TDI;                // Test Data In
     60    input          CDBGPWRUPACK;       // Debug power up acknowledge
     61  
     62    // Miscellaneous
     63    input          PORESETn;           // PowerOn reset
     64    input          SYSRESETn;          // System reset
     65    input          RSTBYPASS;          // Reset Bypass
     66    input          CGBYPASS;           // Architectural clock gate bypass
     67    input          FCLK;               // Free running clock
     68    input          HCLK;               // System clock
     69    input          TRACECLKIN;         // TPIU trace port clock
     70    input          STCLK;              // System Tick clock
     71    input   [25:0] STCALIB;            // System Tick calibration
     72    input   [31:0] AUXFAULT;           // Auxillary FSR pulse inputs
     73    input          BIGEND;             // Static endianess select
     74  
     75    // Interrupt
     76    input  [239:0] INTISR;             // Interrupts
     77    input          INTNMI;             // Non-maskable Interrupt
     78  
     79    // Code (instruction & literal) bus
     80    input          HREADYI;            // ICode-bus ready
     81    input   [31:0] HRDATAI;            // ICode-bus read data
     82    input    [1:0] HRESPI;             // ICode-bus transfer response
     83    input          IFLUSH;             // ICode-bus buffer flush
     84    input          HREADYD;            // DCode-bus ready
     85    input   [31:0] HRDATAD;            // DCode-bus read data
     86    input    [1:0] HRESPD;             // DCode-bus transfer response
     87    input          EXRESPD;            // DCode-bus exclusive response
     88  
     89    // System Bus
     90    input          HREADYS;            // System-bus ready
     91    input   [31:0] HRDATAS;            // System-bus read data
     92    input    [1:0] HRESPS;             // System-bus transfer response
     93    input          EXRESPS;            // System-bus exclusive response
     94  
     95    // Sleep
     96    input          RXEV;               // Wait for exception input
     97    input          SLEEPHOLDREQn;      // Hold core in sleep mode
     98  
     99    // External Debug Request
     100    input          EDBGRQ;             // Debug Request
     101    input          DBGRESTART;         // External Debug Restart request
     102  
     103    // DAP HMASTER override
     104    input          FIXMASTERTYPE;      // Override HMASTER for AHB-AP accesses
     105  
     106    // WIC
     107    input          WICENREQ;           // WIC mode Request from PMU
     108  
     109    // Timestamp intereace
     110    input [47:0]   TSVALUEB;           // Binary coded timestamp value
     111  
     112    // Scan
     113    input          SE;                 // Scan Enable
     114  
     115    // Logic disable
     116    input          MPUDISABLE;         // Disable the MPU (act as default)
     117    input          DBGEN;              // Enable debug
     118    input          NIDEN;              // Enable non-invasive debug
     119  
     120    // Added for DesignStart
     121    // Tie-High if code mux is used
     122    input          DNOTITRANS;         // I/DCode arbitration control
     123  
     124    // Debug
     125    output         TDO;                // Test Data Out
     126    output         nTDOEN;             // Test Data Out Enable
     127    output         CDBGPWRUPREQ;       // Debug power up request
     128  
     129    // Single Wire
     130    output         SWDO;               // SingleWire data out
     131    output         SWDOEN;             // SingleWire output enable
     132    output         JTAGNSW;            // JTAG mode(1) or SW mode(0)
     133  
     134    // Single Wire Viewer
     135    output         SWV;                // SingleWire Viewer Data
     136  
     137    // TracePort Output
     138    output         TRACECLK;           // TracePort clock reference
     139    output   [3:0] TRACEDATA;          // TracePort Data
     140  
     141    // HTM data
     142    output  [31:0] HTMDHADDR;          // HTM data HADDR
     143    output   [1:0] HTMDHTRANS;         // HTM data HTRANS
     144    output   [2:0] HTMDHSIZE;          // HTM data HSIZE
     145    output   [2:0] HTMDHBURST;         // HTM data HBURST
     146    output   [3:0] HTMDHPROT;          // HTM data HPROT
     147    output  [31:0] HTMDHWDATA;         // HTM data HWDATA
     148    output         HTMDHWRITE;         // HTM data HWRITE
     149    output  [31:0] HTMDHRDATA;         // HTM data HRDATA
     150    output         HTMDHREADY;         // HTM data HREADY
     151    output   [1:0] HTMDHRESP;          // HTM data HRESP
     152  
     153    // Code (instruction & literal) bus
     154    output   [1:0] HTRANSI;            // ICode-bus transfer type
     155    output   [2:0] HSIZEI;             // ICode-bus transfer size
     156    output  [31:0] HADDRI;             // ICode-bus address
     157    output   [2:0] HBURSTI;            // ICode-bus burst length
     158    output   [3:0] HPROTI;             // ICode-bus protection
     159    output   [1:0] MEMATTRI;           // ICode-bus memory attributes
     160    output   [1:0] HMASTERD;           // DCode-bus master
     161    output   [1:0] HTRANSD;            // DCode-bus transfer type
     162    output   [2:0] HSIZED;             // DCode-bus transfer size
     163    output  [31:0] HADDRD;             // DCode-bus address
     164    output   [2:0] HBURSTD;            // DCode-bus burst length
     165    output   [3:0] HPROTD;             // DCode-bus protection
     166    output   [1:0] MEMATTRD;           // ICode-bus memory attributes
     167    output         EXREQD;             // ICode-bus exclusive request
     168    output         HWRITED;            // DCode-bus write not read
     169    output  [31:0] HWDATAD;            // DCode-bus write data
     170  
     171    // System Bus
     172    output   [1:0] HMASTERS;           // System-bus master
     173    output   [1:0] HTRANSS;            // System-bus transfer type
     174    output         HWRITES;            // System-bus write not read
     175    output   [2:0] HSIZES;             // System-bus transfer size
     176    output         HMASTLOCKS;         // System-bus lock
     177    output  [31:0] HADDRS;             // System-bus address
     178    output  [31:0] HWDATAS;            // System-bus write data
     179    output   [2:0] HBURSTS;            // System-bus burst length
     180    output   [3:0] HPROTS;             // System-bus protection
     181    output   [1:0] MEMATTRS;           // System-bus memory attributes
     182    output         EXREQS;             // System-bus exclusive request
     183  
     184    // Core Status
     185    output   [3:0] BRCHSTAT;           // Branch status
     186    output         HALTED;             // Core is halted via debug
     187    output         DBGRESTARTED;       // External Debug Restart Ready
     188    output         LOCKUP;             // Lockup indication
     189    output         SLEEPING;           // Core is sleeping
     190    output         SLEEPDEEP;          // System can enter deep sleep
     191    output         SLEEPHOLDACKn;      // Indicate core is force in sleep mode
     192    output   [8:0] ETMINTNUM;          // Interrupt that is currently active
     193    output   [2:0] ETMINTSTAT;         // Interrupt activation status
     194    output         TRCENA;             // Trace Enable
     195    output   [7:0] CURRPRI;            // Current Int Priority
     196  
     197    // Reset request
     198    output         SYSRESETREQ;        // System reset request
     199  
     200    // Events
     201    output         TXEV;               // Event output
     202  
     203    // Clock gating control
     204    output         GATEHCLK;           // when high, HCLK can be turned off
     205  
     206    // WIC
     207    output         WICENACK;           // WIC mode acknowledge from WIC
     208    output         WAKEUP;             // Wake-up request from WIC
