     8  /**
     9   * Abstract:
     10   * ahb_uvc_cfg is used by user to configure the uvc whether uvc agent is
     11   * master or slave and also containes configurations for master and slave
     12   * agents.
     13   *
     14   */
     15  
     16  `ifndef AHB_UVC_CONFIG
     17  `define AHB_UVC_CONFIG
     18  
     19  class ahb_uvc_config extends uvm_object;
     20  
     21    /** configurations which can be modified to select that uvc contain how much
     22     * number of master or slave. 
     23     */
     24    int no_of_masters = 1;
     25    int no_of_slaves  = 1;
     26  
     27    /** master agent and slave agent config handles */
     28    ahb_mas_agent_cfg master_cfg[];
     29    ahb_slv_agent_cfg slave_cfg[];
     30  
     31    /** factory registration */
     32    `uvm_object_utils_begin(ahb_uvc_config)
     33      `uvm_field_int(no_of_masters,UVM_PRINT)
     34      `uvm_field_int(no_of_slaves,UVM_PRINT)
     35      `uvm_field_array_object(master_cfg,UVM_PRINT)
     36      `uvm_field_array_object(slave_cfg,UVM_PRINT)
     37    `uvm_object_utils_end
     38  
     39    /** create_sub_cfgs */
     40    extern function void create_sub_cfgs(int no_of_master=1,int no_of_slave=1);
     41  
     42    /** class constructor */
     43    extern function new(string name = "ahb_uvc_config");
     44  
     45  endclass : ahb_uvc_config
     46  
     47  //*****************************************************************************
     48  //methods
     49  //*****************************************************************************
     50  
     51  /** create_sub_cfgs */
     52  function void ahb_uvc_config::create_sub_cfgs(int no_of_master=1,int no_of_slave=1);
     53    
     54    master_cfg = new[no_of_master];
     55    slave_cfg  = new[no_of_slave];
     56  
     57    foreach(master_cfg[i])
     58      master_cfg[i] = ahb_mas_agent_cfg::type_id::create($sformatf("mas_agent_cfg[%0d]",i));
     59    foreach(slave_cfg[i]) 
     60      slave_cfg[i]  = ahb_slv_agent_cfg::type_id::create($sformatf("slv_agent_cfg[%0d]",i));
     61  
     62  endfunction : create_sub_cfgs
     63  
     64  /** new function */
     65  function ahb_uvc_config::new(string name = "ahb_uvc_config");
     66    super.new(name);
     67  endfunction : new
     68  
     69  `endif  //AHB_UVC_CONFIG
