     64  /**                  RUN PHASE
     65  *
     66  *   DESCRIPTION:  -  In the run phase we have implemented driving logic of all
     67  *   signals which are send to inf/DUT. In the RUN PHASE we have different task
     68  *   function for same type of logic.
     69  *   
     70  *   EXPLANATION OF ALL TASK/FUNCTION:-
     71  *   1)RESET TASK          : -  When active low reset asserted then our all signals are
     72  *                              going into initial state as per specification. 
     73  *   2)ADDR_PHASE TASK     : -  This method is used for driving all addr_phase signal.
     74  *   3)DATA_PHASE TASK     : -  This method is used for driving wr_data to the
     75  *   inf/dut.
     76  *   4)SLV_ERROR  METHOD     : - When slave assert error then our driver call this
     77  *                             task for going into IDLE state at second cycle of error along with this we
     78  *                             are deleting our trans_type queue and data_queue.
     79  *   5)ADDR_INCREMENT METHOD : - Our address is increment with the help of HSIZE value so this
     80  *                             task contains that incrementation of address and for wrapping burst we need to wrap
     81  *                             address at address boundary.
     82  *   6)FLASH_DATA METHOD    : - This method contains, As per specification we need to delete trans and
     83  *                             data queue and disable data_phase to complete transfer(iteration).
     84    * 7)BEFORE GFT_NEXT_ITEM METHOD :- This method used to check previous
     85    * transaction as per spectification and then move ahead for get_next_item.
     86      *
     87  *                  */
     88  
     89  /**   RUN PHASE   */
     90  task ahb_mas_drv::run_phase(uvm_phase phase);
     91    
     92    wait(!mas_vif.hresetn)
     93      reset();
     94    
     95    forever begin
     96      fork
     97        forever begin
     98  	
     99          before_get_next_item();                              //task_method_to check before get_next_item
     100          if(mas_vif.HRESP && (!mas_vif.HREADY))
     101            slv_error_task();
     102   
     103  	else begin
     104  
     105            seq_item_port.get_next_item(req);
     106            rst_flag=1;
     107            `uvm_info("MASTER DRIVER","THIS IS INSIDE MASTER DRIVER",UVM_LOW) 
     108            req.print();
     109            send_to_dut();
     110            seq_item_port.item_done();
     111            size_of_queue=0;
     112            rst_flag=0;
     113  
     114          end
     115  
     116        end
     117         
     118        begin wait(!mas_vif.hresetn); end
     119            
     120    join_any
     121  
     122    disable fork;
     123      
     124    if(!mas_vif.hresetn)
     125      reset();
     126      
     127   end
     128  
     129  endtask : run_phase

     131  /**  reset task  */
     132  task ahb_mas_drv::reset();
     133  
     134    `uvm_info("MASTER DRIVER","THIS IS MASTER SIDE RESET",UVM_LOW)
     135    
     136    `MAS_DRV_CB.HADDR     <= `ADDR_WIDTH'b0;
     137    `MAS_DRV_CB.HTRANS    <= 2'b0;
     138    `MAS_DRV_CB.HBURST    <= 3'b0;
     139    `MAS_DRV_CB.HMASTLOCK <= 1'b0;             
     140    `MAS_DRV_CB.HPROT     <= `HPROT_WIDTH'b0011; 
     141    `MAS_DRV_CB.HWRITE    <= 1'b0; 
     142    `MAS_DRV_CB.HMASTER   <= `HMASTER_WIDTH'b0;              
     143    `MAS_DRV_CB.HNONSEC   <= 1'b0;
     144    `MAS_DRV_CB.HSIZE     <= 3'b0;
     145    `MAS_DRV_CB.HWDATA    <= `DATA_WIDTH'b0;
     146    trans_queue.delete();
     147    temp_queue.delete();
     148       
     149    if(rst_flag==1) begin
     150      seq_item_port.item_done();
     151      rst_flag=0;
     152    end
     153       
     154    @(posedge mas_vif.hresetn);
     155    @(`MAS_DRV_CB);
     156  
     157  endtask : reset

     159  /**  send to_dut method  */
     160  task ahb_mas_drv::send_to_dut();
     161        
     162    fork
     163      addr_phase();
     164      data_phase();
     165    join_any
     166  
     167  endtask : send_to_dut

     169  /**  address phase method  */
     170  task ahb_mas_drv::addr_phase();
     171  
     172    req.calc();
     173  
     174    foreach(req.htrans_type[i])
     175      trans_queue.push_back(req.htrans_type[i]);
     176      
     177    if(req.idle_trans_queue.size>0) begin
     178      foreach(req.idle_trans_queue[i])
     179        trans_queue.push_back(req.idle_trans_queue[i]);
     180      `uvm_info(get_type_name,$sformatf("new transaction queue ===%p",trans_queue),UVM_DEBUG)
     181    end
     182    
     183   /* if(req.busy_trans_queue.size>0 && req.hburst_type!=hburst_enum'(SINGLE)) begin
     184      foreach(req.busy_trans_queue[i])
     185        trans_queue.insert(req.index_for_busy_transfer,req.busy_trans_queue[i]);
     186      `uvm_info(get_type_name,$sformatf("new transaction queue ===%p",trans_queue),UVM_DEBUG)
     187    end*/
     188   
     189    size_of_queue = $size(trans_queue);
     190    `uvm_info("MASTER DRIVER","THIS IS INSIDE ADDR PHASE",UVM_DEBUG)
     191  
     192    if(`MAS_MON_CB.HRESP && !`MAS_MON_CB.HREADY)
     193      slv_error_task();
     194  	
     195    else begin
     196  
     197      if((`MAS_MON_CB.HTRANS!==htrans_enum'(BUSY) && `MAS_MON_CB.HTRANS!==htrans_enum'(IDLE)) && !`MAS_MON_CB.HREADY) begin
     198  
     199        while(!`MAS_MON_CB.HREADY) begin
     200  
     201          @(`MAS_DRV_CB);
     202  
     203          if(`MAS_MON_CB.HRESP && !`MAS_MON_CB.HREADY)
     204            slv_error_task();
     205  	
     206        end
     207  
     208      end
     209  
     210      `MAS_DRV_CB.HWRITE <= req.HWRITE;
     211      if(req.enb_haddr_cross_boundary_cb==1) begin
     212         `uvm_do_callbacks(ahb_mas_drv,driver_callback,unaligned_haddr_error(req)); end
     213  
     214      `MAS_DRV_CB.HADDR    <= req.HADDR;
     215      `MAS_DRV_CB.HBURST   <= hburst_enum'(req.hburst_type);
     216      `MAS_DRV_CB.HTRANS   <= trans_queue.pop_front();
     217      `MAS_DRV_CB.HPROT    <= req.HPROT;
     218      `MAS_DRV_CB.HMASTLOCK<= req.HMASTLOCK;
     219  
     220      if(req.enb_hsize_greater_than_width_cb==1)begin 
     221         `uvm_do_callbacks(ahb_mas_drv,driver_callback,hsize_greater_than_data(req)); 
     222         `MAS_DRV_CB.HSIZE<=hsize_enum'(req.hsize_type);
     223      end
     224      
     225      else
     226        `MAS_DRV_CB.HSIZE  <= hsize_enum'(req.hsize_type);
     227  
     228      `uvm_info("MASTER DRIVER","THIS IS INSIDE FIRST NONSEQ",UVM_DEBUG)
     229          
     230      do begin	    
     231        first_trans_wait = 1;
     232        @(`MAS_DRV_CB);
     233  
     234        if(`MAS_MON_CB.HRESP && !`MAS_MON_CB.HREADY)
     235          slv_error_task();
     236  
     237      end while(!`MAS_MON_CB.HREADY);
     238    
     239     if(req.hburst_type!=hburst_enum'(SINGLE) || req.idle_trans_queue.size>0) begin
     240       for(int i=0;(i<size_of_queue-1);i++) begin
     241   
     242         if(!first_trans_wait)    
     243           @(`MAS_DRV_CB);
     244        
     245         if(first_trans_wait)
     246           first_trans_wait = 0;
     247  
     248         `uvm_info("MASTER DRIVER",$sformatf("THIS IS AFTER FIRST TRANSECTION OF TRANS TYPE -- %0d",i),UVM_DEBUG)
     249  
     250         if(`MAS_MON_CB.HRESP) begin
     251           `uvm_info("MASTER DRIVER","THIS IS IN ERROR SCENARION AFTER FIRST TRANSFER IF NOT SINGLE",UVM_DEBUG)
     252           slv_error_task();
     253           break;
     254          					          
     255          end
     256  
     257          else begin
     258            if(req.enb_hsize_change_in_between_burst_cb==1) begin
     259              `MAS_DRV_CB.HSIZE<=hsize_enum'($clog2(`DATA_WIDTH/8));
     260            end
     261            if(req.enb_hburst_change_in_between_burst_cb==1) begin
     262              `MAS_DRV_CB.HBURST<=hburst_enum'($urandom_range(0,7));
     263            end
     264           
     265            if(req.enb_x_value_cb==1) begin
     266              x_value_signal();
     267            end
     268  
     269            `uvm_info("MASTER DRIVER","THIS IS IN OKAY SCENARIO AFTER FIRST TRANSFER",UVM_HIGH)
     270         
     271            if(`MAS_MON_CB.HTRANS==htrans_enum'(BUSY) && !`MAS_MON_CB.HREADY) begin
     272  
     273              `uvm_info("MASTER DRIVER","THIS IS INSIDE BUSY WAITED SCENARIO",UVM_DEBUG)
     274            
     275              if(trans_queue[0]==SEQ && req.hburst_type!=SINGLE) begin
     276  
     277                `uvm_info("MASTER DRIVER","THIS IS INSIDE BUSY WAITED AND TRANS CHANGED TO SEQ FOR FIXED",UVM_DEBUG)
     278              
     279                `MAS_DRV_CB.HTRANS <= trans_queue.pop_front(); 
     280              
     281  	      do begin
     282  
     283                  @(`MAS_DRV_CB);
     284                  if(`MAS_MON_CB.HRESP && !`MAS_MON_CB.HREADY)
     285                    slv_error_task();
     286  	        
     287  		first_trans_wait = 1;
     288  	   
     289  	      end while(!`MAS_MON_CB.HREADY);
     290  
     291              end
     292  
     293  	    else if(trans_queue[0]==NONSEQ && req.hburst_type==INCR) begin
     294            
     295                `uvm_info("MASTER DRIVER","THIS IS INSIDE UNDEFINED BUSY WAITED CHANGED TO NONSEQ",UVM_DEBUG)
     296                flash_data();
     297                break;
     298  
     299              end
     300  
     301              else if(trans_queue[0]==IDLE && req.hburst_type==INCR) begin
     302  
     303                `uvm_info("MASTER DRIVER","THIS IS INSIDE UNDEFINED BUSY WAITED CHANGED TO IDLE",UVM_DEBUG)
     304              
     305                `MAS_DRV_CB.HTRANS <= trans_queue.pop_front();
     306                `MAS_DRV_CB.HADDR  <= $urandom;	
     307              
     308  	      if(trans_queue[0]!=IDLE) begin	    
     309  
     310                  @(`MAS_DRV_CB);
     311                  flash_data();
     312                  break;
     313  	           
     314                end
     315             
     316              end 
     317  
     318  	    else if(trans_queue[0]==BUSY)
     319                `MAS_DRV_CB.HTRANS <= trans_queue.pop_front();
     320  	    
     321            end
     322  
     323            else if(`MAS_MON_CB.HTRANS==htrans_enum'(IDLE) && !`MAS_MON_CB.HREADY) begin
     324  
     325              `uvm_info("MASTER DRIVER","THIS IS INSIDE IDLE WAITED",UVM_DEBUG)
     326  
     327              if(trans_queue[0]==NONSEQ) begin
     328   
     329                `uvm_info("MASTER DRIVER","THIS IS INSIDE IDLE WAITED CHANGED TO NONSEQ",UVM_DEBUG)
     330  	      flash_data();
     331                break;
     332  
     333              end
     334  
     335              else if (trans_queue[0]==IDLE)begin   //changed 25-01
     336  
     337                `uvm_info("MASTER DRIVER","THIS IS INSIDE IDLE WAITED CHANGED TO IDLE",UVM_DEBUG)
     338                `MAS_DRV_CB.HTRANS <= trans_queue.pop_front();
     339                `MAS_DRV_CB.HADDR  <= $urandom;
     340                @(`MAS_DRV_CB);
     341                flash_data();
     342  	    
     343  	    end	  
     344  
     345  	    else begin
     346  
     347                `uvm_info("MASTER DRIVER","THIS IS INSIDE SEQ OR BUSY AFTER IDLE TRANSECTION",UVM_DEBUG)
     348                `MAS_DRV_CB.HTRANS <= trans_queue.pop_front();
     349  
     350              end
     351  	
     352  	  end
     353  
     354            else if(trans_queue[0]==BUSY && `MAS_MON_CB.HREADY) begin
     355   
     356              `uvm_info("MASTER DRIVER","THIS IS INSIDE BUSY TRANSECTION",UVM_DEBUG)
     357              
     358  	    `MAS_DRV_CB.HTRANS <= trans_queue.pop_front();
     359  	        
     360              if(`MAS_MON_CB.HTRANS!=BUSY)
     361                addr_increment();       
     362          
     363            end
     364          
     365            else if(trans_queue[0]==IDLE && `MAS_MON_CB.HREADY) begin
     366  
     367              `uvm_info("MASTER DRIVER","THIS IS INSIDE IDLE TRANSECTION",UVM_DEBUG)
     368            
     369              `MAS_DRV_CB.HTRANS <= trans_queue.pop_front();
     370               `MAS_DRV_CB.HADDR  <= $urandom;
     371               if(trans_queue[0]!=IDLE) begin
     372                 flash_data();
     373                 break;
     374               end
     375  
     376             end
     377          
     378             else begin
     379            
     380               `uvm_info("MASTER DRIVER","THIS IS INSIDE SEQ TRANSFER",UVM_DEBUG)
     381            
     382               if(`MAS_MON_CB.HTRANS!=htrans_enum'(BUSY)) begin
     383              
     384                 `uvm_info("MASTER DRIVER","THIS IS SEQ IF PREV IS BUSY SO ADDR NOT CHANGE",UVM_DEBUG)
     385  
     386                 while(!`MAS_MON_CB.HREADY) begin
     387  
     388                   @(`MAS_DRV_CB);
     389  
     390                   if(`MAS_MON_CB.HRESP && !`MAS_MON_CB.HREADY) begin
     391                 
     392                     slv_error_task();
     393  	           break;
     394                      	
     395                   end
     396  
     397                 end
     398  
     399                 addr_increment();
     400           
     401               end
     402           
     403  	     if(trans_queue[0]==NONSEQ) begin
     404  
     405                 `MAS_DRV_CB.HTRANS <= trans_queue.pop_front();
     406                 flash_data();
     407                 break;
     408  	        
     409               end
     410        
     411               else begin
     412                 `MAS_DRV_CB.HTRANS <= trans_queue.pop_front();
     413               end
     414  
     415             end
     416        
     417           end
     418  
     419         end
     420       end
     421       `uvm_info(get_type_name,"ADDR PHASE!!going to be end!!",UVM_LOW)
     422     end
     423  endtask : addr_phase

     425  /**  data phase method  */
     426  task ahb_mas_drv::data_phase();
     427    
     428    if(req.HWRITE) begin
     429      
     430      foreach(req.HWDATA[i])
     431        temp_queue.push_back(req.HWDATA[i]);
     432        
     433    end
     434  
     435    `uvm_info("MASTER DRIVER","THIS IS INSIDE DATA PHASE",UVM_HIGH)
     436  
     437    for(int i=0; i<req.no_of_beats;i++)begin
     438    
     439      if(req.HWRITE) begin 
     440  
     441        `uvm_info("MASTER DRIVER","THIS IS INSIDE WRITE SCENARIO DATA PHASE",UVM_HIGH)
     442        
     443        @(`MAS_DRV_CB);          
     444        `uvm_info("MASTER DRIVER",$sformatf("THIS IS LOOP_COUNT -- %0d",i),UVM_HIGH)
     445        
     446        if(`MAS_MON_CB.HRESP==OKAY) begin
     447  
     448          `uvm_info("MASTER DRIVER","THIS IS INSIDE WRITE SCENARIO OKAY RESP DATA PHASE",UVM_DEBUG)
     449          foreach(temp_queue[i])
     450            `uvm_info("MASTER DRIVER",$sformatf("THIS IS TRANS -- %p",temp_queue[i]),UVM_DEBUG)
     451                        
     452          wait(`MAS_MON_CB.HREADY);  
     453  
     454          if(`MAS_MON_CB.HTRANS==htrans_enum'(BUSY) || `MAS_MON_CB.HTRANS==htrans_enum'(IDLE)) begin
     455          
     456            i--;      
     457  
     458          end
     459  
     460          else begin
     461           	        	         	  	          
     462            `uvm_info("MASTER DRIVER","THIS IS INSIDE DATA SENDING TO THE INTERFACE",UVM_HIGH)
     463            `MAS_DRV_CB.HWDATA<=temp_queue.pop_front();
     464         
     465          end
     466  
     467        end
     468  
     469        else begin
     470           
     471          `uvm_info("MASTER DRIVER","THIS IS INSIDE WRITE SCENARIO ERROR RESP DATA PHASE",UVM_HIGH)		
     472          break;
     473        
     474        end
     475  
     476      end
     477  
     478      else begin 
     479                
     480        `uvm_info("MASTER DRIVER","THIS IS INSIDE READ SCENARIO DATA PHASE",UVM_HIGH)
     481  	
     482        @(`MAS_DRV_CB);
     483  
     484        `uvm_info("MASTER DRIVER",$sformatf("THIS IS LOOP_COUNT -- %0d",i),UVM_DEBUG)
     485        
     486        wait(`MAS_MON_CB.HREADY);      
     487        
     488        if(`MAS_MON_CB.HTRANS==htrans_enum'(BUSY) || `MAS_MON_CB.HTRANS==htrans_enum'(IDLE)) begin
     489    
     490          i--;
     491  
     492        end	
     493                
     494      end
     495      
     496    end
     497    `uvm_info(get_type_name,$sformatf("DATA PHASE!!size of queue==========%p",trans_queue),UVM_LOW)
     498  
     499    //repeat(req.no_of_wait) begin
     500    repeat(req.idle_trans_queue.size()) begin
     501     `uvm_info(get_type_name,$sformatf("value of wait=%0d",req.no_of_wait),UVM_LOW)
     502      @(`MAS_DRV_CB);
     503    end
     504    `uvm_info(get_type_name,"DATA PHASE!!going to be end!!",UVM_LOW)
     505  endtask : data_phase

     530  /** before get_next_item method  */
     531  task ahb_mas_drv::before_get_next_item(); 
     532          if(!((`MAS_MON_CB.HTRANS===htrans_enum'(BUSY) || `MAS_MON_CB.HTRANS===htrans_enum'(IDLE)) && !`MAS_MON_CB.HREADY && !`MAS_MON_CB.HRESP) && !first_trans_wait) begin
     533  
     534            `uvm_info("MASTER DRIVER","PREVIOUS TRANS IS NOT BUSY",UVM_HIGH)
     535             @(`MAS_DRV_CB);
     536  
     537  	        if(!`MAS_MON_CB.HRESP && `MAS_MON_CB.HTRANS!==htrans_enum'(IDLE)) begin
     538              
     539  	          while(!`MAS_MON_CB.HREADY) begin
     540  
     541                @(`MAS_DRV_CB);
     542  
     543                if(`MAS_MON_CB.HRESP && !`MAS_MON_CB.HREADY)
     544                   slv_error_task();
     545                              
     546              end
     547  
     548            end
     549  
     550          end
     551  
     552  	      if(first_trans_wait)
     553            first_trans_wait = 0;
     554  
     555  endtask

     557  /**   slv_error_task  */
     558  function void ahb_mas_drv::slv_error_task();

     560    `MAS_DRV_CB.HTRANS <= htrans_enum'(IDLE);
     561    `MAS_DRV_CB.HADDR  <= $urandom;
     562    temp_queue.delete();
     563    trans_queue.delete();
     564    disable data_phase;
     565  
     566  endfunction : slv_error_task

     568  /**   addr increment function */
     569  function void ahb_mas_drv::addr_increment();
     570  
     571    req.HADDR += req.addr_incr;		  
     572    `MAS_DRV_CB.HADDR <= req.HADDR;
     573    
     574    if((req.HADDR==req.higher_boundary) && ((req.hburst_type==WRAP4) || (req.hburst_type==WRAP8) || (req.hburst_type==WRAP16))) begin
     575  
     576      `uvm_info("MASTER DRIVER","THIS IS SEQ IF PREV IS BUSY SO ADDR NOT CHANGE WRAP BOUNDRY",UVM_MEDIUM)
     577
