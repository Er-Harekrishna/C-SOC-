     8  /**
     9   * Abstract:
     10   * ahb_mas_agent_cfg is used by user to configure the agents whether our agent
     11   * is active or passive and also used for enabling and disabling checkers by
     12   * the use of uvm_active_passive_enum and uvm_checker_enable_disable_enum.
     13   *
     14   */
     15  
     16  `ifndef AHB_MASTER_AGENT_CONFIG
     17  `define AHB_MASTER_AGENT_CONFIG
     18  
     19  class ahb_mas_agent_cfg extends uvm_object;
     20  
     21    /** configurations which can be modified 
     22     *  is_active - if UVM_ACTIVE configures the agent as active if its UVM_PASSIVE
     23     *              configures the agent as passive.
     24     *  is_enable - if ENABLE the checker is enables and if DISABLE checker is disabled.
     25     */
     26    uvm_active_passive_enum         is_active = UVM_ACTIVE;
     27    uvm_checker_enable_disable_enum is_enable = ENABLE;
     28  
     29    /** factory registration */
     30    `uvm_object_utils_begin(ahb_mas_agent_cfg)
     31  
     32      `uvm_field_enum(uvm_active_passive_enum, is_active, UVM_PRINT)
     33      `uvm_field_enum(uvm_checker_enable_disable_enum, is_enable, UVM_PRINT)
     34  
     35    `uvm_object_utils_end
     36  
     37    /** class constructor */
     38    extern function new(string name = "ahb_mas_agent_config");
     39  
     40  endclass : ahb_mas_agent_cfg
     41  
     42  //*****************************************************************************
     43  //methods
     44  //*****************************************************************************
     45  
     46  /** class constructor */
     47  function ahb_mas_agent_cfg::new(string name = "ahb_mas_agent_config");
     48    super.new(name);
     49  endfunction : new
     50  
     51  `endif  //AHB_MASTER_AGENT_CONFIG
