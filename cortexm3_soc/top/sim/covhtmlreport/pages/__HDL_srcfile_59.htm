     25  //new function
     26  function ahb_system_config::new(string name="ahb_system_config");
     27  
     28    super.new(name);
     29  
     30    env_cfg = ahb_env_config::type_id::create($sformatf("env_cfg"));   
     31    
     32    env_cfg.has_scoreboard = 0;
     33    env_cfg.has_coverage = 0;
     34    env_cfg.no_of_uvc = 1;
     35    env_cfg.create_sub_cfgs(1);
     36    //env_cfg.uvc_cfg[0].no_of_masters = 1;
     37    //env_cfg.uvc_cfg[0].no_of_slaves  = 0;
     38    env_cfg.uvc_cfg[0].no_of_masters = 5;
     39    env_cfg.uvc_cfg[0].no_of_slaves  = 5;
     40    
     41    env_cfg.uvc_cfg[0].create_sub_cfgs(5,5);
     42    
     43    env_cfg.uvc_cfg[0].master_cfg[0].is_active = UVM_PASSIVE;
     44    env_cfg.uvc_cfg[0].master_cfg[1].is_active = UVM_PASSIVE;
     45    env_cfg.uvc_cfg[0].master_cfg[2].is_active = UVM_PASSIVE;
     46    env_cfg.uvc_cfg[0].master_cfg[3].is_active = UVM_PASSIVE;
     47    env_cfg.uvc_cfg[0].master_cfg[4].is_active = UVM_PASSIVE;
     48    
     49    env_cfg.uvc_cfg[0].master_cfg[0].is_enable = DISABLE;
     50    env_cfg.uvc_cfg[0].master_cfg[1].is_enable = DISABLE;
     51    env_cfg.uvc_cfg[0].master_cfg[2].is_enable = DISABLE;
     52    env_cfg.uvc_cfg[0].master_cfg[3].is_enable = DISABLE;
     53    env_cfg.uvc_cfg[0].master_cfg[4].is_enable = DISABLE;
     54    
     55    env_cfg.uvc_cfg[0].slave_cfg[0].is_active = UVM_PASSIVE;
     56    env_cfg.uvc_cfg[0].slave_cfg[1].is_active = UVM_PASSIVE;
     57    env_cfg.uvc_cfg[0].slave_cfg[2].is_active = UVM_PASSIVE;
     58    env_cfg.uvc_cfg[0].slave_cfg[3].is_active = UVM_PASSIVE;
     59    env_cfg.uvc_cfg[0].slave_cfg[4].is_active = UVM_PASSIVE;
     60  
     61    env_cfg.print();
     62  
     63  endfunction : new
