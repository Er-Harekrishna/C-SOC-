     8  /**
     9   * Abstract:
     10   * ahb_base_master_sequence provides API methods and seuqence length and
     11   * p_sequencer handle and every sequences are extends from this sequence.
     12   * This class defines a api medthods to provide ease for making test cases.
     13   *
     14   * Execution phase: main_phase
     15   * Sequencer: ahb_mas_seqr
     16   *
     17   */
     18  
     19  `ifndef AHB_BASE_MASTER_SEQUENCE
     20  `define AHB_BASE_MASTER_SEQUENCE
     21  
     22  class ahb_base_master_sequence extends uvm_sequence #(ahb_mas_trans);
     23  
     24    /** factroy registration */
     25    `uvm_object_param_utils(ahb_base_master_sequence)
     26   
     27    /** number of transactions that will be generated */
     28    int sequence_length = 20;
     29  
     30    /** p_sequencer declartion */
     31    `uvm_declare_p_sequencer(ahb_mas_seqr)
     32   
     33    /** class constructor */
     34    extern function new(string name="ahb_base_master_sequence");
     35  
     36    /** body method for generating stimulus */
     37    extern task pre_body();
     38    extern task post_body();
     39  
     40    /** API methods for user convinience in making sequences */
     41    extern task write(hburst_enum burst,hsize_enum size,htrans_enum trans[$]={},int incr_length=1);
     42    extern task read(hburst_enum burst,htrans_enum trans[$]={},int incr_length=1);
     43   
     44  endclass : ahb_base_master_sequence
     45  
     46  //*****************************************************************************
     47  //methods
     48  //*****************************************************************************
     49  
     50  /** class constructor */
     51  function ahb_base_master_sequence::new(string name="ahb_base_master_sequence");
     52    super.new(name);
     53  endfunction : new
     54  
     55  /**pre body */
     56  task ahb_base_master_sequence::pre_body();
     57  
     58    if(starting_phase !=null)
     59      starting_phase.raise_objection(this);
     60  
     61    else
     62     `uvm_error("seq","error")
     63          
     64  endtask : pre_body
     65  
     66  /** post body*/
     67  task ahb_base_master_sequence::post_body();
     68  
     69    if(starting_phase !=null) begin
     70      starting_phase.drop_objection(this);
     71      starting_phase.phase_done.set_drain_time(this,500ns);
     72    end
     73  
     74    else
     75     `uvm_error("seq","error")
     76           
     77  endtask : post_body
     78  
     79  /** task write */
     80  task ahb_base_master_sequence::write(hburst_enum burst,hsize_enum size,htrans_enum trans[$]={},int incr_length=1);
     81  
     82    req = ahb_mas_trans::type_id::create("req");  //create the req (seq item)
     83    wait_for_grant();                            //wait for grant
     84    assert(req.randomize() with {HWRITE==1; hburst_type==burst; hsize_type==size; incr_size==incr_length;});
     85    if($size(trans)!=0)
     86       req.htrans_type=trans;
     87    send_request(req);                           //send req to driver
     88    wait_for_item_done();                        //wait for item done from driver
     89  	
     90  endtask :write
     91  
     92  /** task read */
     93  task ahb_base_master_sequence::read(hburst_enum burst,htrans_enum trans[$]={},int incr_length=1);
     94  
     95    req = ahb_mas_trans::type_id::create("req");  //create the req (seq item)
     96    wait_for_grant();                            //wait for grant
     97    assert(req.randomize() with {HWRITE==0; hburst_type==burst; incr_size==incr_length;});
     98    if($size(trans)!=0)
     99       req.htrans_type=trans;
     100    send_request(req);                           //send req to driver
     101    wait_for_item_done();                        //wait for item done from driver
     102  	
     103  endtask :read
     104  
     105  `endif // AHB_BASE_MASTER_SEQUENCE
