1  parameter WIDTH=32;
     2  parameter SIZE=4096;
     3  parameter DEPTH=SIZE/WIDTH;
     4  parameter ADDR_WIDTH = $clog2(DEPTH);
     5  module apb_memory(
     6  	clk_i, rst_i, addr_i, wr_rd_i, wdata_i,sel_i, rdata_o, valid_i, ready_o
     7  );
     8  
     9  
     10  input clk_i, rst_i, wr_rd_i,sel_i;
     11  input [ADDR_WIDTH-1:0] addr_i;
     12  input [WIDTH-1:0] wdata_i;
     13  output reg [WIDTH-1:0] rdata_o;
     14  input valid_i;
     15  output reg ready_o;
     16  
     17  //declare the storage
     18  reg [WIDTH-1:0] mem [DEPTH-1:0];
     19  integer i;
     20  
     21  always @(posedge clk_i) begin
     22  if(sel_i) begin
     23  	if (rst_i == 1) begin
     24  		ready_o = 0;
     25  		rdata_o = 0;
     26  		for (i = 0; i < DEPTH; i=i+1) begin
     27  			mem[i] = 0;
     28  		end
     29  	end
     30  	else begin
     31  		if (valid_i == 1) begin
     32  			ready_o = 1;
     33  			if (wr_rd_i) begin
     34  				mem[addr_i] = wdata_i;
     35  			end
     36  			else begin //read
     37  				rdata_o = mem[addr_i];
     38  			end
     39  		end
     40  		else begin
     41  			ready_o = 0;
     42  		end
     43  	end
     44  end
     45  else begin
     46  
     47  end
     48  end
     49  endmodule
