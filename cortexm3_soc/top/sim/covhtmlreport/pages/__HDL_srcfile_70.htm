1  module ahb_slave(HADDR,HWDATA,HTRANS,HREADYin,HWRITE,HRESP, HRDATA,
     2  		HSIZE,HCLK,HRESETn,HADDR_1,HWDATA_1,HADDR_2,HWDATA_2,
     3  		HADDR_3,HWDATA_3,HWRITEreg,valid,TEMP_SEL,PRDATA);
     4  
     5  input	[31:0]HADDR,HWDATA,PRDATA;
     6  input	[1:0]HTRANS;
     7  input	HREADYin,HWRITE;
     8  input	[2:0]HSIZE;
     9  input	HCLK,HRESETn;
     10  	
     11  
     12  output reg	[31:0]HADDR_1,
     13  		HWDATA_1,
     14  		HADDR_2,
     15  		HWDATA_2,
     16  		HADDR_3,
     17  		HWDATA_3;
     18  
     19  output 		[31:0]HRDATA;
     20  	
     21  
     22  output reg 	HWRITEreg,valid;
     23  output reg	[1:0]HRESP;
     24  output reg	[2:0]TEMP_SEL;
     25  
     26  
     27  
     28  assign HRDATA=PRDATA;
     29  
     30  
     31  parameter IDLE=2'b00,
     32  	  BUSY=2'b01,
     33  	  NONSEQ=2'b10,
     34  	  SEQ=2'b11;
     35  
     36  
     37  
     38  always@(posedge HCLK,negedge HRESETn)
     39    begin
     40  	HRESP=0;
     41  
     42  	if(~HRESETn)
     43         	 begin
     44            HADDR_1<=0;
     45  	  HADDR_2<=0;
     46  	  HADDR_3<=0;
     47  	 end
     48  
     49  	
     50  	else
     51  	  begin
     52  	    HADDR_1<=HADDR;
     53  	    HADDR_2<=HADDR_1;
     54  	    HADDR_3<=HADDR_2;
     55  	  end
     56   end
     57  
     58  
     59  always@(posedge HCLK, negedge HRESETn)
     60    begin
     61  	if(~HRESETn)
     62  	  begin
     63  	    HWDATA_1<=0;
     64  	    HWDATA_2<=0;
     65  	    HWDATA_3<=0;
     66  	  end
     67  
     68  	else
     69  	  begin
     70  	   HWDATA_1<=HWDATA;
     71  	   HWDATA_2<=HWDATA_1;
     72  	   HWDATA_3<=HWDATA_2;
     73  	  end
     74    end
     75  
     76  
     77  always@(posedge HCLK, negedge HRESETn)
     78    begin
     79  	if(~HRESETn)
     80  	   HWRITEreg<=0;
     81  	else
     82  	   HWRITEreg<=HWRITE;
     83    end
     84         
     85  
     86  always@(*)
     87    begin
     88      if(~HRESETn)
     89         valid=0;
     90      
     91      else if((HADDR>=32'h4000_0000 && HADDR<= 32'h4000_FFFF) /*&&  (HTRANS==NONSEQ || HTRANS==SEQ)*/)
     92       
     93           valid=1'b1;
     94      else
     95           valid=1'b0;
     96     end
     97  
     98  
     99  always@(*)
     100    begin
     101  
     102     
     103           if((HADDR>=32'h4000_2000 && HADDR<= 32'h4000_2FFF)) 
     104  		TEMP_SEL=3'b100;
     105  	 else if((HADDR>=32'h4000_1000 && HADDR<= 32'h4000_1FFF))
     106  		TEMP_SEL=3'b010;
     107   	 else if((HADDR>=32'h4000_0000 && HADDR<= 32'h4000_0FFF))
     108  		TEMP_SEL=3'b001;
     109  	//secure purpose , modified
     110  	//else TEMP_SEL=3'b000;
     111  		
     112  	 
     113    end
     114  endmodule
