     8  /**
     9   * Abstract:
     10   * Class ahb_slave_mem_response_min_wait_sequence defines a sequence class that
     11   * the testbench uses to provide slave response with zero wait cycles. 
     12   * The sequence receives a response object of type ahb_slv_trans from 
     13   * slave sequencer.The sequence class then randomizes the response with 
     14   * constraints and provides it to the slave driver within the slave agent. 
     15   * The sequence also writes into or reads from the slave memory.
     16   *
     17   * Execution phase: run_phase
     18   * Sequencer: ahb_slv_seqr
     19   */
     20  
     21  `ifndef AHB_SLAVE_MEM_RESPONSE_MIN_WAIT_SEQUENCE
     22  `define AHB_SLAVE_MEM_RESPONSE_MIN_WAIT_SEQUENCE
     23  
     24  class ahb_slave_mem_response_min_wait_sequence extends ahb_base_slave_sequence;
     25  
     26    /** factroy registration */
     27    `uvm_object_param_utils(ahb_slave_mem_response_min_wait_sequence)
     28  
     29    /** class constructor */
     30    extern function new(string name="ahb_slave_mem_response_min_wait_sequence");
     31   
     32    /** body method */
     33    extern task body();
     34     
     35  endclass : ahb_slave_mem_response_min_wait_sequence
     36  
     37  //*****************************************************************************
     38  //methods
     39  //*****************************************************************************
     40  
     41  /** class constructor */
     42  function ahb_slave_mem_response_min_wait_sequence::new(string name="ahb_slave_mem_response_min_wait_sequence");
     43    super.new(name);
     44  endfunction : new
     45  
     46  /** body method */
     47  task ahb_slave_mem_response_min_wait_sequence::body();
     48          
     49    forever begin
     50      
     51      p_sequencer.item_collected_fifo.get(req);
     52    	 
     53      `uvm_info("SLAVE SEQUENCE","This Is Slave Sequence Collected Transaction",UVM_MEDIUM)
     54      req.print();
     55  
     56      write(req);
     57      read(req);
     58      assert(req.randomize() with { hresp_type == OKAY; HREADYOUT==1; wait_cycles==0; });
     59  
     60      `uvm_send(req);
     61   
     62    end   
     63       
     64  endtask : body
     65  
     66  `endif //AHB_SLAVE_MEM_RESPONSE_MIN_WAIT_SEQUENCE
