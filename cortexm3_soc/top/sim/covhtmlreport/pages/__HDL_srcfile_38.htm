     8  /**
     9   * Abstract:
     10   * This class defines a sequence in which a single burst random AHB WRITE followed 
     11   * by a random AHB READ sequence is generated using `uvm_do_with macros.
     12   *
     13   * Execution phase: main_phase
     14   * Sequencer: ahb_mas_seqr
     15   *
     16   */
     17  
     18  `ifndef AHB_MASTER_SANITY_SEQUENCE
     19  `define AHB_MASTER_SANITY_SEQUENCE
     20  
     21  class ahb_master_sanity_sequence extends ahb_base_master_sequence;
     22  
     23    /** factroy registration */
     24    `uvm_object_param_utils(ahb_master_sanity_sequence)
     25      
     26    /** class constructor */
     27    extern function new(string name="ahb_master_sanity_sequence");
     28  
     29    /** body method for generating stimulus */
     30    extern task body();
     31   
     32  endclass : ahb_master_sanity_sequence
     33  
     34  //*****************************************************************************
     35  //methods
     36  //*****************************************************************************
     37  
     38  /** class constructor */
     39  function ahb_master_sanity_sequence::new(string name="ahb_master_sanity_sequence");
     40    super.new(name);
     41  endfunction : new
     42  
     43  /** body method */ 
     44  task ahb_master_sanity_sequence::body();
     45   
     46    `uvm_info("body", "Entered ...", UVM_MEDIUM)
     47  
     48    if(!(uvm_config_db #(int unsigned)::get(null, get_full_name(), "sequence_length", sequence_length)))
     49      `uvm_info("BODY","GET FAILED",UVM_MEDIUM)
     50  
     51    `uvm_info("body", $sformatf("sequence_length is %0d", sequence_length), UVM_MEDIUM);
     52    
     53    for(int i=0;i<=$clog2(`DATA_WIDTH/8);i++) begin
     54      repeat (sequence_length) begin
     55       
     56        `uvm_do_with(req,
     57         {  HADDR       == 0;
     58            HWRITE      == 1;
     59            hburst_type == hburst_enum'(SINGLE);
     60            hsize_type  == hsize_enum'(WORD);
     61            incr_size   == 1;
     62         })
     63  
     64        `uvm_do_with(req,
     65         {  HADDR       == 0;
     66            HWRITE      == 0;
     67            hburst_type == hburst_enum'(SINGLE);
     68            hsize_type  == hsize_enum'(HALFWORD);
     69            incr_size   == 1;
     70         })
     71      
     72      end   
     73    end
     74  
     75    `uvm_info("body", "Exiting...", UVM_MEDIUM)
     76    
     77  endtask : body
     78  
     79  `endif // AHB_MASTER_SANITY_SEQUENCE
