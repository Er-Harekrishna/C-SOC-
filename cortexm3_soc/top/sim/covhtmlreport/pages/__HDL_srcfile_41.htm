     8  /**
     9   * Abstract:
     10   * This class defines a sequence in which a all incremenr burst like 
     11   * incr, incr4 , incr8 , incr16 random AHB WRITE followed by a random AHB READ
     12   * sequence is generated using `uvm_do_with macros.
     13   *
     14   * Execution phase: main_phase
     15   * Sequencer: ahb_mas_seqr
     16   *
     17   */
     18  
     19  `ifndef AHB_MASTER_INCREMENT_SEQUENCE
     20  `define AHB_MASTER_INCREMENT_SEQUENCE
     21  
     22  class ahb_master_incr_sequence extends ahb_base_master_sequence;
     23  
     24    /** factroy registration */
     25    `uvm_object_utils(ahb_master_incr_sequence)
     26   
     27    /** class constructor */
     28    extern function new(string name="ahb_master_incr_sequence");
     29    
     30    /** body method for generating stimulus */
     31    extern task body();
     32  
     33  endclass : ahb_master_incr_sequence
     34  
     35  //*****************************************************************************
     36  //methods
     37  //*****************************************************************************
     38  
     39  /** class constructor */
     40  function ahb_master_incr_sequence::new(string name="ahb_master_incr_sequence");
     41    super.new(name);
     42  endfunction 
     43  
     44  /** body method for generating stimulus */
     45  task ahb_master_incr_sequence::body();
     46    
     47    `uvm_info("body", "Entered ...", UVM_MEDIUM)
     48  
     49    if(!(uvm_config_db #(int unsigned)::get(null, get_full_name(), "sequence_length", sequence_length)))
     50      `uvm_info("BODY","GET FAILED",UVM_MEDIUM)
     51  
     52    `uvm_info("body", $sformatf("sequence_length is %0d", sequence_length), UVM_MEDIUM);
     53   
     54    for(int k=1;k<8;k=k+2) begin
     55      for(int i=0;i<=$clog2(`DATA_WIDTH/8);i++) begin
     56        repeat (sequence_length) begin
     57  
     58          `uvm_do_with(req,
     59           {  HADDR       == 2**(i);
     60  	    HWRITE      == 1;
     61  	    hburst_type == hburst_enum'(k);
     62  	    hsize_type  == hsize_enum'(i);
     63  	    incr_size   == 1;
     64  	    foreach(busy_trans_cycles[i])
     65  	      busy_trans_cycles[i]==0;
     66  	 })
     67          
     68          `uvm_do_with(req,
     69           {  HADDR       == 2**(i);
     70  	    HWRITE      == 0;
     71  	    hburst_type == hburst_enum'(k);
     72  	    hsize_type  == hsize_enum'(i);
     73  	    incr_size   == 1;
     74  	    foreach(busy_trans_cycles[i])
     75  	      busy_trans_cycles[i]==0;
     76         	 })
     77  
     78        end   
     79      end
     80    end
     81   
     82    `uvm_do_with(req,
     83     {  HADDR       == 1020;
     84        HWRITE      == 1;
     85        hburst_type == hburst_enum'(INCR4);
     86        hsize_type  == hsize_enum'(WORD);
     87        foreach(busy_trans_cycles[i])
     88          busy_trans_cycles[i]==0;
     89     })
     90  
     91    `uvm_do_with(req,
     92     {  HADDR       == 1020;
     93        HWRITE      == 0;
     94        hburst_type == hburst_enum'(INCR4);
     95        hsize_type  == hsize_enum'(HALFWORD);
     96        foreach(busy_trans_cycles[i])
     97          busy_trans_cycles[i]==0;
     98     })
     99  
     100    `uvm_do_with(req,
     101     {  HADDR       == 6;
     102        HWRITE      == 1;
     103        hburst_type == hburst_enum'(INCR4);
     104        hsize_type  == hsize_enum'(HALFWORD);
     105        foreach(busy_trans_cycles[i])
     106          busy_trans_cycles[i]==0;
     107     })
     108  
     109    `uvm_do_with(req,
     110     {  HADDR       == 6;
     111        HWRITE      == 0;
     112        hburst_type == hburst_enum'(INCR4);
     113        hsize_type  == hsize_enum'(HALFWORD);
     114        foreach(busy_trans_cycles[i])
     115          busy_trans_cycles[i]==0;
     116     })
     117  
     118    `uvm_info("body", "Exiting ...", UVM_MEDIUM)
     119  
     120  endtask : body
     121  
     122  `endif //AHB_MASTER_INCREMENT_SEQUENCE
