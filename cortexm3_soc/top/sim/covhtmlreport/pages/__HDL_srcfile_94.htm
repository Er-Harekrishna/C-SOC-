     167      wire  [2:0] burst_op2;     // Port 2 HBURST signal

     179      wire [31:0] auser_op3;     // Port 3 HAUSER signal
     180      wire  [1:0] trans_op3;     // Port 3 HTRANS signal
     181      wire        write_op3;     // Port 3 HWRITE signal
     182      wire  [2:0] size_op3;      // Port 3 HSIZE signal
     183      wire  [2:0] burst_op3;     // Port 3 HBURST signal
     184      wire  [3:0] prot_op3;      // Port 3 HPROT signal

     223

     225      reg   [1:0] i_htransm;     // Internal HTRANSM
     226      reg   [2:0] i_hburstm;     // Internal HBURSTM

     239    assign req_port2 = held_tran_op2 & sel_op2;
     240    assign req_port3 = held_tran_op3 & sel_op3;
     241  
     242    // Arbiter instance for resolving requests to this output stage
     243    cmsdk_MyArbiterNameM2 u_output_arb (
     244  
     245      .HCLK       (HCLK),
     246      .HRESETn    (HRESETn),
     247  
     248      .req_port2   (req_port2),
     249      .req_port3   (req_port3),
     250  
     251      .HREADYM    (i_hreadymuxm),
     252      .HSELM      (i_hselm),
     253      .HTRANSM    (i_htransm),
     254      .HBURSTM    (i_hburstm),
     255      .HMASTLOCKM (hlock_arb),
     256  
     257      .addr_in_port (addr_in_port),
     258      .no_port     (no_port)
     259  
     260      );

     263    // Active signal combinatorial decode
     264    always @ (addr_in_port or no_port)
     265      begin : p_active_comb
     266        // Default value(s)
     267        active_op2 = 1'b0;
     268        active_op3 = 1'b0;
     269  
     270        // Decode selection when enabled
     271        if (~no_port)
     272          case (addr_in_port)
     273            3'b010 : active_op2 = 1'b1;
     274            3'b011 : active_op3 = 1'b1;
     275            default : begin
     276              active_op2 = 1'bx;
     277              active_op3 = 1'bx;
     278            end
     279          endcase // case(addr_in_port)
     280      end // block: p_active_comb
     281  
     282  
     283    //  Address/control output decode
     284    always @ (
     285               sel_op2 or addr_op2 or trans_op2 or write_op2 or
     286               size_op2 or burst_op2 or prot_op2 or
     287               auser_op2 or
     288               master_op2 or mastlock_op2 or
     289               sel_op3 or addr_op3 or trans_op3 or write_op3 or
     290               size_op3 or burst_op3 or prot_op3 or
     291               auser_op3 or
     292               master_op3 or mastlock_op3 or
     293               addr_in_port or no_port
     294             )
     295      begin : p_addr_mux
     296        // Default values
     297        i_hselm     = 1'b0;
     298        HADDRM      = {32{1'b0}};
     299        HAUSERM     = {32{1'b0}};
     300        i_htransm   = 2'b00;
     301        HWRITEM     = 1'b0;
     302        HSIZEM      = 3'b000;
     303        i_hburstm   = 3'b000;
     304        HPROTM      = {4{1'b0}};
     305        HMASTERM    = 4'b0000;
     306        i_hmastlockm= 1'b0;
     307  
     308        // Decode selection when enabled
     309        if (~no_port)
     310          case (addr_in_port)
     311            // Bus-switch input 2
     312            3'b010 :
     313              begin
     314                i_hselm     = sel_op2;
     315                HADDRM      = addr_op2;
     316                HAUSERM     = auser_op2;
     317                i_htransm   = trans_op2;
     318                HWRITEM     = write_op2;
     319                HSIZEM      = size_op2;
     320                i_hburstm   = burst_op2;
     321                HPROTM      = prot_op2;
     322                HMASTERM    = master_op2;
     323                i_hmastlockm= mastlock_op2;
     324              end // case: 4'b010
     325  
     326            // Bus-switch input 3
     327            3'b011 :
     328              begin
     329                i_hselm     = sel_op3;
     330                HADDRM      = addr_op3;
     331                HAUSERM     = auser_op3;
     332                i_htransm   = trans_op3;
     333                HWRITEM     = write_op3;
     334                HSIZEM      = size_op3;
     335                i_hburstm   = burst_op3;
     336                HPROTM      = prot_op3;
     337                HMASTERM    = master_op3;
     338                i_hmastlockm= mastlock_op3;
     339              end // case: 4'b011
     340  
     341            default :
     342              begin
     343                i_hselm     = 1'bx;
     344                HADDRM      = {32{1'bx}};
     345                HAUSERM     = {32{1'bx}};
     346                i_htransm   = 2'bxx;
     347                HWRITEM     = 1'bx;
     348                HSIZEM      = 3'bxxx;
     349                i_hburstm   = 3'bxxx;
     350                HPROTM      = {4{1'bx}};
     351                HMASTERM    = 4'bxxxx;
     352                i_hmastlockm= 1'bx;
     353              end // case: default
     354          endcase // case(addr_in_port)
     355      end // block: p_addr_mux
     356  
     357    // hsel_lock provides support for AHB masters that address other
     358    // slave regions in the middle of a locked sequence (i.e. HSEL is
     359    // de-asserted during the locked sequence).  Unless HMASTLOCK is
     360    // held during these intermediate cycles, the OutputArb scheme will
     361    // lose track of the locked sequence and may allow another input
     362    // port to access the output port which should be locked
     363    assign next_hsel_lock = (i_hselm & i_htransm[1] & i_hmastlockm) ? 1'b1 :
     364                           (i_hmastlockm == 1'b0) ? 1'b0 :
     365                            hsel_lock;
     366  
     367    // Register hsel_lock
     368    always @ (negedge HRESETn or posedge HCLK)
     369      begin : p_hsel_lock
     370        if (~HRESETn)
     371          hsel_lock <= 1'b0;
     372        else
     373          if (i_hreadymuxm)
     374            hsel_lock <= next_hsel_lock;
     375      end
     376  
     377    // Version of HMASTLOCK which is masked when not selected, unless a
     378    // locked sequence has already begun through this port
     379    assign hlock_arb = i_hmastlockm & (hsel_lock | i_hselm);
     380  
     381    assign HTRANSM    = i_htransm;
     382    assign HBURSTM    = i_hburstm;
     383    assign HSELM      = i_hselm;
     384    assign HMASTLOCKM = i_hmastlockm;
     385  
     386    // Dataport register
     387    always @ (negedge HRESETn or posedge HCLK)
     388      begin : p_data_in_port_reg
     389        if (~HRESETn)
     390          data_in_port <= {3{1'b0}};
     391        else
     392          if (i_hreadymuxm)
     393            data_in_port <= addr_in_port;
     394      end
     395  
     396    // HWDATAM output decode
     397    always @ (
     398               wdata_op2 or
     399               wdata_op3 or
     400               data_in_port
     401             )
     402      begin : p_data_mux
     403        // Default value
     404        HWDATAM = {32{1'b0}};
     405  
     406        // Decode selection
     407        case (data_in_port)
     408          3'b010 : HWDATAM  = wdata_op2;
     409          3'b011 : HWDATAM  = wdata_op3;
     410          default : HWDATAM = {32{1'bx}};
     411        endcase // case(data_in_port)
     412      end // block: p_data_mux
     413  
     414    // HWUSERM output decode
     415    always @ (
     416               wuser_op2 or
     417               wuser_op3 or
     418               data_in_port
     419             )
     420      begin : p_wuser_mux
     421        // Default value
     422        HWUSERM  = {32{1'b0}};
     423  
     424        // Decode selection
     425        case (data_in_port)
     426          3'b010 : HWUSERM  = wuser_op2;
     427          3'b011 : HWUSERM  = wuser_op3;
     428          default : HWUSERM  = {32{1'bx}};
     429        endcase // case(data_in_port)
     430      end // block: p_wuser_mux
     431  
     432    // ---------------------------------------------------------------------------
     433    // HREADYMUXM generation
     434    // ---------------------------------------------------------------------------
     435    // The HREADY signal on the shared slave is generated directly from
     436    //  the shared slave HREADYOUTS if the slave is selected, otherwise
     437    //  it mirrors the HREADY signal of the appropriate input port
     438    always @ (negedge HRESETn or posedge HCLK)
     439      begin : p_slave_sel_reg
     440        if (~HRESETn)
     441          slave_sel <= 1'b0;
     442        else
     443          if (i_hreadymuxm)
     444            slave_sel  <= i_hselm;
     445      end
     446  
     447    // HREADYMUXM output selection
     448    assign i_hreadymuxm = (slave_sel) ? HREADYOUTM : 1'b1;
     449  
     450    // Drive output with internal version of the signal
     451    assign HREADYMUXM = i_hreadymuxm;
     452  
     453  
     454  endmodule
